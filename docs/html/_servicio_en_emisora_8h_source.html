15// alReves() utilidad
16// pone al revés el contenido de una array en el mismo array
17// -------------------------------------------------------------------------------------------------------------
18template< typename T >
19T * alReves( T * p, int n ) {
20 T aux;
21
22 for ( int i = 0; i < n / 2; i++ ) {
23 aux = p[i];
24 p[i] = p[n - i - 1];
25 p[n - i - 1] = aux;
26 }
27 return p;
28} // ()

31// pString:<char>, pUint:<N>, tamMax:N --> stringAUint8AlReves() --> pUint:<N>
32// Descripción: Método que nos sirve para cambiar los valores de una lista de tipo String a una lista de
33// números enteros, ambas usando punteros con los que referenciar a las listas y un tamaño máximo
34// -------------------------------------------------------------------------------------------------------------
35uint8_t * stringAUint8AlReves( const char * pString, uint8_t * pUint, int tamMax ) {
36
37 int longitudString = strlen( pString );
38 int longitudCopiar = ( longitudString > tamMax ? tamMax : longitudString );
39 // copio nombreServicio -> uuidServicio pero al revés
40 for ( int i = 0; i <= longitudCopiar - 1; i++ ) {
41 pUint[ tamMax - i - 1 ] = pString[ i ];
42 } // for
43
44 return pUint;
45} // ()
