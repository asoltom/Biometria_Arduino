124 // -------------------------------------------------------------------------------------------------------------
125 //
126 // Ejemplo de Beacon (31 bytes)
127 //
128 // https://os.mbed.com/blog/entry/BLE-Beacons-URIBeacon-AltBeacons-iBeacon/
129 //
130 // The iBeacon Prefix contains the hex data : 0x0201061AFF004C0215. This breaks down as follows:
131 //
132 // 0x020106 defines the advertising packet as BLE General Discoverable and BR/EDR high-speed incompatible.
133 // Effectively it says this is only broadcasting, not connecting.
134 //
135 // 0x1AFF says the following data is 26 bytes long and is Manufacturer Specific Data.
136 //
137 // 0x004C is Apple’s Bluetooth Sig ID and is the part of this spec that makes it Apple-dependent.
138 //
139 // 0x02 is a secondary ID that denotes a proximity beacon, which is used by all iBeacons.
140 //
141 // 0x15 defines the remaining length to be 21 bytes (16+2+2+1).
142 //
143 // Por ejemmplo:
144 //
145 // 1. prefijo: 9bytes
146 // 0x02, 0x01, 0x06, // advFlags 3bytes
147 // 0x1a, 0xff, // advHeader 2 (0x1a = 26 = 25(lenght de 0x4c a 0xca)+1) 0xFF -> BLE_GAP_AD_TYPE_MANUFACTURER_SPECIFIC_DATA
148 // 0x4c, 0x00, // companyID 2bytes
149 // 0x02, // ibeacon type 1 byte
150 // 0x15, // ibeacon length 1 byte (dec=21 lo que va a continuación: desde la 'f' hasta 0x01)
151 //
152 // 2. uuid: 16bytes
153 // 'f', 'i', 's', 't', 'r', 'o', 'f', 'i', 's', 't', 'r', 'o', 0xa7, 0x10, 0x96, 0xe0
154 //
155 // 2 major: 2bytes
156 // 0x04, 0xd2,
157 //
158 // minor: 2bytes
159 // 0x10, 0xe1,
160 //
161 // 0xca, // tx power : 1bytes
162 //
163 // 0x01, // este es el byte 31 = BLE_GAP_ADV_SET_DATA_SIZE_MAX, parece que sobra
164 //
165 // -------------------------------------------------------------------------------------------------------------
166 // Para enviar como carga libre los últimos 21 bytes de un iBeacon
167 // (lo que normalmente sería uuid-16 major-2 minor-2 txPower-1)
168 // -------------------------------------------------------------------------------------------------------------
169 // carga:char, tamanyoCarga:N --> emitirAnuncioIBeaconLibre()
170 // Descripción: Método de la clase EmisoraBLE para enviar un texto
171 // -----------------------------------------------------------------------------------------------------------
172 void emitirAnuncioIBeaconLibre( const char * carga, const uint8_t tamanyoCarga ) {
173
174 (*this).detenerAnuncio();
175
176 Bluefruit.Advertising.clearData();
177 Bluefruit.ScanResponse.clearData(); // hace falta?
178
179 // Bluefruit.setTxPower( (*this).txPower ); creo que no lo pongo porque es uno de los bytes de la parte de carga que utilizo
180 Bluefruit.setName( (*this).nombreEmisora );
181 Bluefruit.ScanResponse.addName();
182
183 Bluefruit.Advertising.addFlags(BLE_GAP_ADV_FLAGS_LE_ONLY_GENERAL_DISC_MODE);
184
185 // -------------------------------------------------------------------------------------------------------------
186 // En principio, habrá puestos los 5 primeros bytes
187 // Falta poner 4 bytes fijos (company ID, beacon type, longitud) y 21 de carga
188 // -------------------------------------------------------------------------------------------------------------
189 uint8_t restoPrefijoYCarga[4 + 21] = {
190 0x4c, 0x00, // companyID 2
191 0x02, // ibeacon type 1byte
192 21, // ibeacon length 1byte (dec=21) longitud del resto // 0x15 // ibeacon length 1byte (dec=21) longitud del resto
193 '-', '-', '-', '-',
194 '-', '-', '-', '-',
195 '-', '-', '-', '-',
196 '-', '-', '-', '-',
197 '-', '-', '-', '-',
198 '-'
199 };
200
201 // -------------------------------------------------------------------------------------------------------------
202 // addData() hay que usarlo sólo una vez. Por eso copio la carga
203 // en el anterior array, donde he dejado 21 sitios libres
204 // -------------------------------------------------------------------------------------------------------------
205 memcpy( &restoPrefijoYCarga[4], &carga[0], ( tamanyoCarga > 21 ? 21 : tamanyoCarga ) );
206
207 // --------------------------
208 // copio la carga para emitir
209 // --------------------------
210 Bluefruit.Advertising.addData( BLE_GAP_AD_TYPE_MANUFACTURER_SPECIFIC_DATA,
211 &restoPrefijoYCarga[0],
212 4 + 21 );
213
214 Bluefruit.Advertising.restartOnDisconnect(true);
215 Bluefruit.Advertising.setInterval(100, 100); // in unit of 0.625 ms
216
217 Bluefruit.Advertising.setFastTimeout( 1 ); // number of seconds in fast mode
218
219 // ---------------------------------------------------------
220 // empieza el anuncio, 0 = tiempo indefinido
221 // ---------------------------------------------------------
222 Bluefruit.Advertising.start( 0 );
223
224 Globales::elPuerto.escribir( "emitiriBeacon libre Bluefruit.Advertising.start( 0 ); \n");
225 } // ()
